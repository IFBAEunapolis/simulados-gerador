\begin{luacode}
function splitgroup(disciplinas)
	for disc in string.gmatch(disciplinas, '[^;]+') do
		local fileHandler = io.open("Atividades/" .. disc .. ".tex","r")
		if fileHandler then
			fileHandler:close()
			tex.print("\\shufflegroup{" .. disc .. "}")
			tex.print("\\insertgroup{" .. disc .. "}")
		end
	end	
end
\end{luacode}
